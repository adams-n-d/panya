01501                               bool single_minus_longopt, bool print_errors, int min_abbr_len)
01502 {

01504   if (args == 0)
01505     numargs = 0;
01506 
01507   int nonops = 0;
01508 
01509   while (numargs != 0 && *args != 0)
01510   {
01511     const char* param = *args; // param can be --long-option, -srto or non-option argument
01512 
01513     // in POSIX mode the first non-option argument terminates the option list
01514     // a lone minus character is a non-option argument
01515     if (param[0] != '-' || param[1] == 0)
01516     {
01517       if (gnu)
01518       {
01519         ++nonops;
01520         ++args;
01521         if (numargs > 0)
01522           --numargs;
01523         continue;
01524       }
01525       else
01526         break;
01527     }
01528 
01529     // -- terminates the option list. The -- itself is skipped.
01530     if (param[1] == '-' && param[2] == 0)
01531     {
01532       shift(args, nonops);
01533       ++args;
01534       if (numargs > 0)
01535         --numargs;
01536       break;
01537     }
01538 
01539     bool handle_short_options;
01540     const char* longopt_name;
01541     if (param[1] == '-') // if --long-option
01542     {
01543       handle_short_options = false;
01544       longopt_name = param + 2;
01545     }
01546     else
01547     {
01548       handle_short_options = true;
01549       longopt_name = param + 1; //for testing a potential -long-option
01550     }
01551 
01552     bool try_single_minus_longopt = single_minus_longopt;
01553     bool have_more_args = (numargs > 1 || numargs < 0); // is referencing argv[1] valid?
01554 
01555     do // loop over short options in group, for long options the body is executed only once
01556     {
01557       int idx;
01558 
01559       const char* optarg;
01560 
01561       /******************** long option **********************/
01562       if (handle_short_options == false || try_single_minus_longopt)
01563       {
01564         idx = 0;
01565         while (usage[idx].longopt != 0 && !streq(usage[idx].longopt, longopt_name))
01566           ++idx;
01567 
01568         if (usage[idx].longopt == 0 && min_abbr_len > 0) // if we should try to match abbreviated long options
01569         {
01570           int i1 = 0;
01571           while (usage[i1].longopt != 0 && !streqabbr(usage[i1].longopt, longopt_name, min_abbr_len))
01572             ++i1;
01573           if (usage[i1].longopt != 0)
01574           { // now test if the match is unambiguous by checking for another match
01575             int i2 = i1 + 1;
01576             while (usage[i2].longopt != 0 && !streqabbr(usage[i2].longopt, longopt_name, min_abbr_len))
01577               ++i2;
01578 
01579             if (usage[i2].longopt == 0) // if there was no second match it's unambiguous, so accept i1 as idx
01580               idx = i1;
01581           }
01582         }
01583 
01584         // if we found something, disable handle_short_options (only relevant if single_minus_longopt)
01585         if (usage[idx].longopt != 0)
01586           handle_short_options = false;
01587 
01588         try_single_minus_longopt = false; // prevent looking for longopt in the middle of shortopt group
01589 
01590         optarg = longopt_name;
01591         while (*optarg != 0 && *optarg != '=')
01592           ++optarg;
01593         if (*optarg == '=') // attached argument
01594           ++optarg;
01595         else
01596           // possibly detached argument
01597           optarg = (have_more_args ? args[1] : 0);
01598       }
01599 
01600       /************************ short option ***********************************/
01601       if (handle_short_options)
01602       {
01603         if (*++param == 0) // point at the 1st/next option character
01604           break; // end of short option group
01605 
01606         idx = 0;
01607         while (usage[idx].shortopt != 0 && !instr(*param, usage[idx].shortopt))
01608           ++idx;
01609 
01610         if (param[1] == 0) // if the potential argument is separate
01611           optarg = (have_more_args ? args[1] : 0);
01612         else
01613           // if the potential argument is attached

01615       }

01620       {
01621         // look for dummy entry (shortopt == "" and longopt == "") to use as Descriptor for unknown options
01622         idx = 0;
01623         while (usage[idx].shortopt != 0 && (usage[idx].shortopt[0] != 0 || usage[idx].longopt[0] != 0))
01624           ++idx;
01625         descriptor = (usage[idx].shortopt == 0 ? 0 : &usage[idx]);
01626       }
01627 
01628       if (descriptor != 0)
01629       {
01630         Option option(descriptor, param, optarg);
01631         switch (descriptor->check_arg(option, print_errors))
01632         {
01633           case ARG_ILLEGAL:
01634             return false; // fatal
01635           case ARG_OK:
01636             // skip one element of the argument vector, if it's a separated argument
01637             if (optarg != 0 && have_more_args && optarg == args[1])
01638             {
01639               shift(args, nonops);
01640               if (numargs > 0)
01641                 --numargs;
01642               ++args;
01643             }
01644 
01645             // No further short options are possible after an argument
01646             handle_short_options = false;
01647 
01648             break;
01649           case ARG_IGNORE:
01650           case ARG_NONE:
01651             option.arg = 0;
01652             break;
01653         }
01654 
01655         if (!action.perform(option))
01656           return false;
01657       }
01658 
01659     } while (handle_short_options);
01660 
01661     shift(args, nonops);
01662     ++args;
01663     if (numargs > 0)
01664       --numargs;
01665 
01666   } // while
01667 
01668   if (numargs > 0 && *args == 0) // It's a bug in the caller if numargs is greater than the actual number
01669     numargs = 0; // of arguments, but as a service to the user we fix this if we spot it.
01670 
01671   if (numargs < 0) // if we don't know the number of remaining non-option arguments
01672   { // we need to count them
01673     numargs = 0;
01674     while (args[numargs] != 0)
01675       ++numargs;
01676   }
01677 
01678   return action.finished(numargs + nonops, args - nonops);
01679 }

01929     void update_length()
01930     {
01931       screenlen = 0;
01932       for (len = 0; ptr[len] != 0 && ptr[len] != '\v' && ptr[len] != '\t' && ptr[len] != '\n'; ++len)
01933       {
01934         ++screenlen;
01935         unsigned ch = (unsigned char) ptr[len];
01936         if (ch > 0xC1) // everything <= 0xC1 (yes, even 0xC1 itself) is not a valid UTF-8 start byte
01937         {
01938           // int __builtin_clz (unsigned int x)
01939           // Returns the number of leading 0-bits in x, starting at the most significant bit
01940           unsigned mask = (unsigned) -1 >> __builtin_clz(ch ^ 0xff);
01941           ch = ch & mask; // mask out length bits, we don't verify their correctness
01942           while (((unsigned char) ptr[len + 1] ^ 0x80) <= 0x3F) // while next byte is continuation byte
01943           {
01944             ch = (ch << 6) ^ (unsigned char) ptr[len + 1] ^ 0x80; // add continuation to char code
01945             ++len;
01946           }
01947           // ch is the decoded unicode code point
01948           if (ch >= 0x1100 && isWideChar(ch)) // the test for 0x1100 is here to avoid the function call in the Latin case
01949             ++screenlen;
01950         }
01951       }
01952     }

02231     {

02251

02259     {
02260       if (wrote_something) // if we already wrote something, we need to start a new line
02261       {
02262         write("\n", 1);
02263         int _ = 0;
02264         indent(write, _, x);
02265       }
02266 
02267       if (!buf_empty())
02268       {

02283     {
02284       if (buf_empty())
02285         return;
02286       int _ = 0;

02288       wrote_something = false;
02289       while (!buf_empty())
02290         write_one_line(write);
02291       write("\n", 1);
02292     }
02293 
02312     void process(IStringWriter& write, const char* data, int len)
02313     {
02314       wrote_something = false;
02315 
02316       while (len > 0)
02317       {
02318         if (len <= width) // quick test that works because utf8width <= len (all wide chars have at least 2 bytes)
02319         {
02320           output(write, data, len);
02321           len = 0;
02322         }
02323         else // if (len > width)  it's possible (but not guaranteed) that utf8len > width
02324         {
02325           int utf8width = 0;
02326           int maxi = 0;
02327           while (maxi < len && utf8width < width)
02328           {
02329             int charbytes = 1;
02330             unsigned ch = (unsigned char) data[maxi];
02331             if (ch > 0xC1) // everything <= 0xC1 (yes, even 0xC1 itself) is not a valid UTF-8 start byte
02332             {
02333               // int __builtin_clz (unsigned int x)
02334               // Returns the number of leading 0-bits in x, starting at the most significant bit
02335               unsigned mask = (unsigned) -1 >> __builtin_clz(ch ^ 0xff);
02336               ch = ch & mask; // mask out length bits, we don't verify their correctness
02337               while ((maxi + charbytes < len) && //
02338                   (((unsigned char) data[maxi + charbytes] ^ 0x80) <= 0x3F)) // while next byte is continuation byte
02339               {
02340                 ch = (ch << 6) ^ (unsigned char) data[maxi + charbytes] ^ 0x80; // add continuation to char code
02341                 ++charbytes;
02342               }
02343               // ch is the decoded unicode code point
02344               if (ch >= 0x1100 && isWideChar(ch)) // the test for 0x1100 is here to avoid the function call in the Latin case
02345               {
02346                 if (utf8width + 2 > width)
02347                   break;
02348                 ++utf8width;
02349               }
02350             }
02351             ++utf8width;
02352             maxi += charbytes;
02353           }
02354 
02355           // data[maxi-1] is the last byte of the UTF-8 sequence of the last character that fits
02356           // onto the 1st line. If maxi == len, all characters fit on the line.
02357 
02358           if (maxi == len)
02359           {
02360             output(write, data, len);
02361             len = 0;
02362           }
02363           else // if (maxi < len)  at least 1 character (data[maxi] that is) doesn't fit on the line
02364           {
02365             int i;
02366             for (i = maxi; i >= 0; --i)
02367               if (data[i] == ' ')
02368                 break;
02369 
02370             if (i >= 0)
02371             {
02372               output(write, data, i);
02373               data += i + 1;
02374               len -= i + 1;
02375             }
02376             else // did not find a space to split at => split before data[maxi]
02377             { // data[maxi] is always the beginning of a character, never a continuation byte
02378               output(write, data, maxi);
02379               data += maxi;
02380               len -= maxi;
02381             }
02382           }
02383         }
02384       }
02385       if (!wrote_something) // if we didn't already write something to make space in the buffer
02386         write_one_line(write); // write at most one line of actual output
02387     }
02388 
02395     LineWrapper(int x1, int x2) :
02396         x(x1), width(x2 - x1), head(0), tail(bufmask)
02397     {
02398       if (width < 2) // because of wide characters we need at least width 2 or the code breaks
02399         width = 2;
02400     }

02408   static void printUsage(IStringWriter& write, const Descriptor usage[], int width = 80, //
02409                          int last_column_min_percent = 50, int last_column_own_line_max_percent = 75)
02410   {
02411     if (width < 1) // protect against nonsense values
02412       width = 80;
02413 
02414     if (width > 10000) // protect against overflow in the following computation
02415       width = 10000;
02416 
02417     int last_column_min_width = ((width * last_column_min_percent) + 50) / 100;
02418     int last_column_own_line_max_width = ((width * last_column_own_line_max_percent) + 50) / 100;
02419     if (last_column_own_line_max_width == 0)
02420       last_column_own_line_max_width = 1;
02421 
02422     LinePartIterator part(usage);
02423     while (part.nextTable())
02424     {
02425 
02426       /***************** Determine column widths *******************************/
02427 
02428       const int maxcolumns = 8; // 8 columns are enough for everyone
02429       int col_width[maxcolumns];
02430       int lastcolumn;
02431       int leftwidth;
02432       int overlong_column_threshold = 10000;
02433       do
02434       {
02435         lastcolumn = 0;
02436         for (int i = 0; i < maxcolumns; ++i)
02437           col_width[i] = 0;
02438 
02439         part.restartTable();
02440         while (part.nextRow())
02441         {
02442           while (part.next())
02443           {
02444             if (part.column() < maxcolumns)
02445             {
02446               upmax(lastcolumn, part.column());
02447               if (part.screenLength() < overlong_column_threshold)
02448                 // We don't let rows that don't use table separators (\t or \v) influence
02449                 // the width of column 0. This allows the user to interject section headers
02450                 // or explanatory paragraphs that do not participate in the table layout.
02451                 if (part.column() > 0 || part.line() > 0 || part.data()[part.length()] == '\t'
02452                     || part.data()[part.length()] == '\v')
02453                   upmax(col_width[part.column()], part.screenLength());
02454             }
02455           }
02456         }
02457 
02458         /*
02459          * If the last column doesn't fit on the same
02460          * line as the other columns, we can fix that by starting it on its own line.
02461          * However we can't do this for any of the columns 0..lastcolumn-1.
02462          * If their sum exceeds the maximum width we try to fix this by iteratively
02463          * ignoring the widest line parts in the width determination until
02464          * we arrive at a series of column widths that fit into one line.
02465          * The result is a layout where everything is nicely formatted
02466          * except for a few overlong fragments.
02467          * */
02468 
02469         leftwidth = 0;
02470         overlong_column_threshold = 0;
02471         for (int i = 0; i < lastcolumn; ++i)
02472         {
02473           leftwidth += col_width[i];
02474           upmax(overlong_column_threshold, col_width[i]);
02475         }
02476 
02477       } while (leftwidth > width);
02478 
02479       /**************** Determine tab stops and last column handling **********************/
02480 
02481       int tabstop[maxcolumns];
02482       tabstop[0] = 0;
02483       for (int i = 1; i < maxcolumns; ++i)
02484         tabstop[i] = tabstop[i - 1] + col_width[i - 1];
02485 
02486       int rightwidth = width - tabstop[lastcolumn];
02487       bool print_last_column_on_own_line = false;
02488       if (rightwidth < last_column_min_width && rightwidth < col_width[lastcolumn])
02489       {
02490         print_last_column_on_own_line = true;
02491         rightwidth = last_column_own_line_max_width;
02492       }
02493 
02494       // If lastcolumn == 0 we must disable print_last_column_on_own_line because
02495       // otherwise 2 copies of the last (and only) column would be output.
02496       // Actually this is just defensive programming. It is currently not
02497       // possible that lastcolumn==0 and print_last_column_on_own_line==true
02498       // at the same time, because lastcolumn==0 => tabstop[lastcolumn] == 0 =>
02499       // rightwidth==width => rightwidth>=last_column_min_width  (unless someone passes
02500       // a bullshit value >100 for last_column_min_percent) => the above if condition
02501       // is false => print_last_column_on_own_line==false
02502       if (lastcolumn == 0)
02503         print_last_column_on_own_line = false;
02504 
02505       LineWrapper lastColumnLineWrapper(width - rightwidth, width);
02506       LineWrapper interjectionLineWrapper(0, width);
02507 
02508       part.restartTable();
02509 
02510       /***************** Print out all rows of the table *************************************/
02511 
02512       while (part.nextRow())
02513       {
02514         int x = -1;
02515         while (part.next())
02516         {
02517           if (part.column() > lastcolumn)
02518             continue; // drop excess columns (can happen if lastcolumn == maxcolumns-1)
02519 
02520           if (part.column() == 0)
02521           {
02522             if (x >= 0)
02523               write("\n", 1);
02524             x = 0;
02525           }
02526 
02527           indent(write, x, tabstop[part.column()]);
02528 
02529           if ((part.column() < lastcolumn)
02530               && (part.column() > 0 || part.line() > 0 || part.data()[part.length()] == '\t'
02531                   || part.data()[part.length()] == '\v'))
02532           {
02533             write(part.data(), part.length());
02534             x += part.screenLength();
02535           }
02536           else // either part.column() == lastcolumn or we are in the special case of
02537                // an interjection that doesn't contain \v or \t
02538           {
02539             // NOTE: This code block is not necessarily executed for
02540             // each line, because some rows may have fewer columns.
02541 
02542             LineWrapper& lineWrapper = (part.column() == 0) ? interjectionLineWrapper : lastColumnLineWrapper;
02543 
02544             if (!print_last_column_on_own_line)
02545               lineWrapper.process(write, part.data(), part.length());
02546           }
02547         } // while
02548 
02549         if (print_last_column_on_own_line)
02550         {
02551           part.restartRow();
02552           while (part.next())
02553           {
02554             if (part.column() == lastcolumn)
02555             {
02556               write("\n", 1);
02557               int _ = 0;
02558               indent(write, _, width - rightwidth);
02559               lastColumnLineWrapper.process(write, part.data(), part.length());
02560             }
02561           }
02562         }
02563 
02564         write("\n", 1);
02565         lastColumnLineWrapper.flush(write);
02566         interjectionLineWrapper.flush(write);
02567       }
02568     }
02569   }
02570 
02571 }
